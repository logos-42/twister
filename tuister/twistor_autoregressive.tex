\section{Foundations of Twistor Theory}

\subsection{Twistor Space Definition}

\begin{definition}[Twistor Space]
The four-dimensional complex twistor space $\mathbb{T}$ is defined as:
\begin{equation}
\mathbb{T} = \mathbb{C}^4 \cong \mathbb{S}_A \oplus \mathbb{S}^{A'}
\end{equation}
where $\mathbb{S}_A \cong \mathbb{C}^2$ is the unprimed spinor space and $\mathbb{S}^{A'} \cong \mathbb{C}^2$ is the primed spinor space.
\end{definition}

\begin{definition}[Twistor]
A twistor $Z^\alpha \in \mathbb{T}$ can be represented as a two-spinor:
\begin{equation}
Z^\alpha = (\omega^A, \pi_{A'})
\end{equation}
where $\omega^A \in \mathbb{S}^A$, $\pi_{A'} \in \mathbb{S}_{A'}$ satisfy the twistor equation:
\begin{equation}\label{eq:twistor_eq}
\omega^A = i x^{AA'} \pi_{A'}
\end{equation}
Here $x^{AA'} = \frac{1}{\sqrt{2}} \sigma_\mu^{AA'} x^\mu$ is the spinor representation of spacetime point $x^\mu$.
\end{definition}

\subsection{Twistor Inner Product and Norm}

\begin{definition}[Twistor Inner Product]
Define the symplectic inner product on twistor space:
\begin{equation}\label{eq:twistor_inner}
\langle Z_1, Z_2 \rangle = \omega_1^A \pi_{2A} + \omega_2^A \pi_{1A} = Z_1^\alpha \bar{Z}_{2\alpha}
\end{equation}
where index raising and lowering use the Levi-Civita symbols $\varepsilon_{AB}$ and $\varepsilon_{A'B'}$.
\end{definition}

\begin{lemma}[Null Twistor Condition]
When $\langle Z, Z \rangle = 0$, we call $Z$ a null twistor. In this case, there exists a spacetime point $x^\mu$ such that equation \eqref{eq:twistor_eq} holds.
\end{lemma}

\subsection{Autoregressive Decomposition}

\begin{definition}[Autoregressive Probability Decomposition]
For a sequence $\mathbf{x} = (x_1, x_2, \ldots, x_T)$, an autoregressive model decomposes its joint probability as:
\begin{equation}\label{eq:autoregressive}
p(\mathbf{x}) = \prod_{t=1}^{T} p(x_t | x_{<t}) = \prod_{t=1}^{T} p(x_t | x_1, \ldots, x_{t-1})
\end{equation}
\end{definition}

\subsection{Attention Mechanism}

\begin{definition}[Scaled Dot-Product Attention]
Given query matrix $Q \in \mathbb{R}^{T \times d_k}$, key matrix $K \in \mathbb{R}^{T \times d_k}$, and value matrix $V \in \mathbb{R}^{T \times d_v}$:
\begin{equation}\label{eq:attention}
\Attention(Q, K, V) = \softmax\left(\frac{QK^T}{\sqrt{d_k}}\right) V
\end{equation}
where $\softmax$ is applied row-wise.
\end{definition}

\begin{definition}[Causal Mask]
Autoregressive attention uses a lower triangular mask $M$:
\begin{equation}\label{eq:causal_mask}
M_{ij} = \begin{cases}
0 & i \geq j \\
-\infty & i < j
\end{cases}
\end{equation}
ensuring that position $i$ can only attend to positions $j \leq i$.
\end{definition}

\end{equation}
\end{enumerate}
\end{theorem}

\subsection{Proof}

\begin{proof}

\textbf{Step 1: Spinor-Vector Correspondence}

Map spinor components to query and key vectors:
\begin{equation}
\omega_i^A \mapsto Q_i, \quad \pi_{iA'} \mapsto K_i
\end{equation}

Using the spinor inner product formula, for $Z_i = (\omega_i^A, \pi_{iA'})$ and $Z_j = (\omega_j^A, \pi_{jA'})$:
\begin{align}
\langle Z_i, Z_j \rangle &= \omega_i^A \pi_{jA} + \omega_j^A \pi_{iA} \\
&= \varepsilon_{AB} \omega_i^A \pi_j^B + \varepsilon_{AB} \omega_j^A \pi_i^B
\end{align}

\textbf{Step 2: Hermitian Structure}

Define the Hermitian metric on twistor space:
\begin{equation}
h(Z_i, Z_j) = \omega_i^A \bar{\omega}_{jA} + \pi_{iA'} \bar{\pi}_j^{A'}
\end{equation}

This corresponds to the attention weight matrix:
\begin{equation}
A_{ij} = \frac{\exp(Q_i K_j^T / \sqrt{d_k})}{\sum_k \exp(Q_i K_k^T / \sqrt{d_k})}
\end{equation}

\textbf{Step 3: Causal Structure from Twistor Equation}

The twistor equation $\omega_i^A = i x_i^{AA'} \pi_{iA'}$ implies that $\omega_i^A$ depends on $\pi_{iA'}$ through the spacetime point $x_i$.

In the autoregressive setting, this dependence maps to:
\begin{equation}
Q_i = f(x_{\leq i}), \quad K_j = g(x_{\leq j})
\end{equation}

The causal constraint $j \leq i$ in the attention mechanism corresponds to the chronological ordering in twistor space.

\textbf{Step 4: Softmax as Twistor Geometry}

Consider the exponential map in twistor space:
\begin{equation}
\exp(\langle Z_i, Z_j \rangle / \lambda)
\end{equation}

where $\lambda = \sqrt{d_k}$ plays the role of "temperature." The softmax normalization:
\begin{equation}
\sum_j A_{ij} = \sum_j \frac{\exp(\langle Z_i, Z_j \rangle / \lambda)}{\sum_k \exp(\langle Z_i, Z_k \rangle / \lambda)} = 1
\end{equation}

corresponds to the projective structure of null twistors on the celestial sphere.

\textbf{Step 5: Multi-Head Attention as Twistor Flag}

Multi-head attention with $h$ heads corresponds to the flag manifold structure:
\begin{equation}
\mathcal{F} = \{(L_1 \subset L_2 \subset \cdots \subset L_h = \mathbb{C}^{d_k}) : \dim L_i = i \cdot d_k/h\}
\end{equation}

Each attention head computes a partial twistor correspondence on a subspace.

Thus, the structure-preserving map $\Phi$ is established.
\end{proof}
